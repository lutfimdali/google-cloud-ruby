def all
  while next?
    next_records = self.next
    push(*next_records)
    self.token = next_records.token
  end
  self
end

def next
  return nil unless next?
  ensure_service!
  grpc = @service.list_metrics token: token
  self.class.from_grpc grpc, @service
rescue GRPC::BadStatus => e
  raise Gcloud::Error.from_error(e)
end
