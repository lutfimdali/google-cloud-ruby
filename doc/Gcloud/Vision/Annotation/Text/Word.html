Overview

Word

A word within a detected text (OCR). See Gcloud::Vision::Annotation::Text.

Examples:

require "gcloud"

gcloud = Gcloud.new
vision = gcloud.vision

image = vision.image "path/to/text.png"
text = image.text

words = text.words
words.count #=> 28

word = words.first
word.text #=> "Google"
word.bounds.count #=> 4
word.bounds.first #=> #<Vertex (x: 13, y: 8)>

def bounds
  return [] unless @gapi["boundingPoly"]
  @bounds ||= Array(@gapi["boundingPoly"]["vertices"]).map do |v|
    Vertex.from_gapi v
  end
end
